52 {
53  std::stringstream ss; //stream to hold the command to be sent.
54 
55  if(!command.getNumArgs())
56  ss << command.getId() << cmd_separator_; // if there are no arguments to be sent, the stream receives only the command id and the command separator.
57  else
58  {
59  //otherwise, if there are some arguments, we will have to iterate over the std::vector args_
60  ss << command.getId() << field_separator_;
61 
62  for(unsigned long i=0; i<command.args_.size() - 1; i++){ //Note that we go from i = 0, to i = args_.size() - 1, because after the last argument a cmd_separator must be added.
63  ss << command.args_[i] << field_separator_; //adds an argument and the field separator to the stream.
64  }
65 
66  ss << command.args_.back() << cmd_separator_; //finishes the command. Now it is ready to be sent.
67 
68  } // else
69 
70  serial_port_.write(ss.str()); //sends the command!
71 
72  if(ack){ // if ack is required
73  return CmdMessenger::waitCmd(ack_id, simple_timeout); //Verifies if the command arrived or not and returns true if received (false otherwise).
74  }
