50 {

52 
53  if(!command.getNumArgs())
54  ss << command.getId() << cmd_separator_; // if there are no arguments to be sent, the stream receives only the command id and the command separator.
55  else
56  {
57  //otherwise, if there are some arguments, we will have to iterate over the std::vector args_

59 
60  for(unsigned long i=0; i<command.args_.size() - 1; i++){ //Note that we go from i = 0, to i = args_.size() - 1, because after the last argument a cmd_separator must be added.
61  ss << command.args_[i] << field_separator_; //adds an argument and the field separator to the stream.
62  }
63 
64  ss << command.args_.back() << cmd_separator_; //finishes the command. Now it is ready to be sent.
65 
66  } // else
67 
68  serial_port_.write(ss.str()); //sends the command!
69 
70  if(ack){ // if ack is required
71  return CmdMessenger::waitCmd(ack_id, simple_timeout); //Verifies if the command arrived or not and returns true if received (false otherwise).
72  }
